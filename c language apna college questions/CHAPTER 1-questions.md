1 ) TO FIND AREA OF SQUARE
```c
#include<stdio.h>
int main() {
    int num;
    printf("enter the length of side:");
    scanf("%d",&num );
    printf("the area of square is : %d",num*num);
    return 0;
}
```
2 ) TO FIND AREA OF CIRCLE
```c
#include<stdio.h>
int main() {
    int num;
    float pi = 3.14;
    printf("enter the radius of circle:");
    scanf("%d",&num );
    printf("the area of circle is : %f",pi*num*num);
    return 0;
}
```
